Let’s model cars!

mtcars_lm <- lm(mpg ~ cyl + disp + hp + drat + wt + qsec + vs + am + gear + carb,
                data = mtcars, y = TRUE)

predicted_mtcars <- mtcars %>%
  mutate(car = rownames(mtcars))
predicted_mtcars$predicted <- mtcars_lm$fitted.values

Let’s look at our model!

paste("r² = ", round(summary(mtcars_lm)$adj.r.squared, 3))
## [1] "r² =  0.807"
ggplot(predicted_mtcars,
       aes(x = predicted, y = mpg, label = car)) + 
  geom_smooth(method = "lm") +
  geom_jitter() + 
  scale_x_continuous(name = "Predicted MPG", limits = c(10, 35), breaks = seq(10, 35, 5)) +
  scale_y_continuous(name = "MPG", limits = c(10, 35), breaks = seq(10, 35, 5)) +
  theme_bw()
